00462         class PooledAllocator
00463         {
00464                 /* We maintain memory alignment to word boundaries by requiring that all
00465                     allocations be in multiples of the machine wordsize.  */
00466                 /* Size of machine word in bytes.  Must be power of 2. */
00467                 /* Minimum number of bytes requested at a time from     the system.  Must be multiple of WORDSIZE. */

00472                 void*   loc;      /* Current location in block to next allocate memory. */
00473 
00474                 void internal_init()
00475                 {
00476                         remaining = 0;
00477                         base = NULL;

00501                 void free_all()
00502                 {
00503                         while (base != NULL) {
00504                                 void *prev = *(static_cast<void**>( base)); /* Get pointer to prev block. */
00505                                 ::free(base);
00506                                 base = prev;
00507                         }

00515                 void* malloc(const size_t req_size)
00516                 {
00517                         /* Round size up to a multiple of wordsize.  The following expression
00518                             only works for WORDSIZE that is a power of 2, by masking last bits of
00519                             incremented size to zero.
00520                          */
00521                         const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
00522 
00523                         /* Check whether a new block must be allocated.  Note that the first word
00524                             of a block is reserved for a pointer to the previous block.
00525                          */
00526                         if (size > remaining) {
00527 
00528                                 wastedMemory += remaining;
00529 
00530                                 /* Allocate new storage. */
00531                                 const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
00532                                                         size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
00533 
00534                                 // use the standard C malloc to allocate memory
00535                                 void* m = ::malloc(blocksize);
00536                                 if (!m) {
00537                                         fprintf(stderr,"Failed to allocate memory.\n");
00538                                         return NULL;
00539                                 }
00540 
00541                                 /* Fill first word of new block with pointer to previous block. */
00542                                 static_cast<void**>(m)[0] = base;
00543                                 base = m;
00544 
00545                                 size_t shift = 0;
00546                                 //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
00547 
00548                                 remaining = blocksize - sizeof(void*) - shift;
00549                                 loc = (static_cast<char*>(m) + sizeof(void*) + shift);
00550                         }
00551                         void* rloc = loc;
00552                         loc = static_cast<char*>(loc) + size;
00553                         remaining -= size;
00554 
00555                         usedMemory += size;
00556 
00557                         return rloc;
00558                 }
